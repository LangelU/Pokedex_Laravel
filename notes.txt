Consulta tipos de pokémon
SELECT p.name, t.name as tipoUno,
(SELECT t.name FROM pokemon p
 LEFT JOIN types t 
 ON p.secondary_type = t.id) AS tipoDos
 FROM pokemon p
 LEFT JOIN types t 
 ON p.primary_type = t.id


Consulta todas las habilidades a partir de id
SELECT abilities_ones.name AS a, abilities_twos.name AS b, abilities_threes.name AS c, abilities_fours.name AS d, abilities_fives.name AS e
FROM abilities_ones, abilities_twos, abilities_threes, abilities_fours, abilities_fives
WHERE abilities_ones.id = 1

Consulta para numero, nombre y tipos
SELECT p.number, p.name, pt.name as TipoUNO, st.name AS TipoDOS
FROM pokemon p, primary_types pt, secondary_types st
WHERE p.primary_type = pt.id AND p.secondary_type = st.id

consulta para numero, nombre y tipos semi terminada
SELECT p.number, p.name, pt.name, st.name
FROM pokemon p
JOIN primary_types pt ON p.primary_type = pt.id 
JOIN secondary_types st ON p.secondary_type = st.id

SELECT *,
COALESCE(p.pre_evolution_name, 'No registra') AS pre_Evo_name,
COALESCE(p.pre_evolution_number, '000') AS pre_Evo_number,
COALESCE(p.evolution_name, 'No registra') AS evo_name,
COALESCE(p.evolution_number, '000') AS evo_number,
COALESCE(p.mega_evolution_name, 'No registra') AS mega_Evo_name,
COALESCE(p.mega_evolution_number, '000') AS mega_Evo_number
FROM pokemon p
WHERE p.id = 1
